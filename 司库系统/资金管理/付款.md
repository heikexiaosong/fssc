#### 推送线上支付指令
- cn.ztessc.service.task.JobCenter#pushPaymentInstruction
```sql
--  查询数据库待支付、待发送、未挂起的、期望支付日期是今天的转账支付指令
select * 
  from FMS_PAYMENT_INSTRUCTIONS P 
 where 1=1
   AND (PAYER_BANK_HEAD_CODE = :bankCode or BUSINESS_TYPE = :businessType)
   AND (hang_status = :hangStatus or hang_status is null or hang_status = '' )
   AND payment_status = '10'
   AND send_status = '1'
   AND instruction_type = :instructionType  -- 1： 线上； 3： 线下
   AND enabled_flag = '0'
   AND EXPECT_PAYMENT_DATE <= :expectPaymentDate
   order by last_update_date asc
```
