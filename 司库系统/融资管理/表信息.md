-- 利息计算表
create table if not exists zfs_fms.fms_loan_rate_compute
(
    ID                      varchar(64)    not null comment '主键ID' primary key,
    PROVIDE_HEADER_ID       varchar(255)   null comment '贷款详情ID',
    AMOUNT                  decimal(22, 2) null comment '金额',
    ANNUAL_RATE             decimal(22, 4) null comment '利率',
    BEGIN_DATE              datetime(6)    null comment '起始日期',
    END_DATE                varchar(200)   null comment '结束日期',
    GROUP_ID                varchar(64)    null comment '集团ID',
    EXTEND_S1               varchar(1)     null comment '扩展字段S1',
    EXTEND_S2               varchar(1)     null comment '扩展字段S2',
    EXTEND_S3               varchar(1)     null comment '扩展字段S3',
    EXTEND_S4               varchar(1)     null comment '扩展字段S4',
    EXTEND_S5               varchar(1)     null comment '扩展字段S5',
    PROVIDE_LOAN_ID         varchar(64)    null comment '贷款放款ID',
    REPAY_PRINCIPAL_PARTIAL decimal(22, 2) null comment '还本金额（部分还息）'
)
