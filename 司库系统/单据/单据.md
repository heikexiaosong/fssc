```sql
        , ZBH.DATA_SOURCE, ZBH.BOE_DATE, ZBH.EMPLOYEE_ID, ZBH.CREATE_BY, ZBH.BOE_ABSTRACT

        , ZBH.CA_PLAIN_DATA, ZBH.CA_SIGNED_DATA, ZBH.CA_AUTH_FLAG, ZBA.APPROVER_USER_CODE, ZBA.TASK_ID
        , ZBA.GROUP_ID, ZBH.BOE_TYPE_CODE, ZBH.FMS_COLLECT_TYPE, ZBH.SOURCE_BOE_NO, ZBH.SOURCE_SYSTEM
        , ZBH.PAYER_BANK_HEAD_CODE, ZBH.ACTUAL_PAYMENT_CURRENCY, ZBH.SETTLEMENT_TYPE, ZBH.UNIQUE_BUS_ID
from FMS_PAYMENT_APPLY_APPROVER ZBA, FMS_PAYMENT_APPLY_HEADER ZBH
where ZBA.BOE_ID = ZBH.ID
        and ZBH.BOE_STATUS = '20'
        and ZBH.ENABLED_FLAG = '0'
        and ZBA.APPROVER_USER_CODE = 'GHG001'
        and ZBH.BOE_TYPE_CODE = 'PAYMENT_BOE'
        and ZBA.group_id = '98384a38c6501ea777273132372e0008'
order by ZBH.boe_date desc, ZBA.id desc
fetch first 20 rows only
```
