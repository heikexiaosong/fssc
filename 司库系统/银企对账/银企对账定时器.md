# 第一步，开启唯一对账码和总金额相等的规则; ruleConfig::reconciliationCode == "Y"
# busMap.get(fmsBusinessNo)[busNo]
bankMap = bankList<[busNo[,busNo]...], []>
erpMap  = erpList<reconciliationCode, []>

for (Map.Entry<String, List<FmsErpTransactionEntity>> erpKey : erpKeys) {
    String key = erpKey.getKey();
    # key为空 或者  bankMap没有这个key  continue    
    List<FmsErpTransactionEntity> eList = erpKey.getValue();
    # 企业账是否一条数据， 不是一条 ， continue
   
    # 通过key获取银行账数据
    List<FmsTransactionDetailsHisDTO> bList = bankMap.get(key);
    
    //唯一对账码和金额相等规则匹配表示 true-满足规则 false-不满足规则
    boolean ruleFlag = matchRule(ruleConfig, bList, eList, index);
    {
        # matchRule 实现， 总金额、 交易类型 以及 交易时间 匹配
        String blendNo = DateUtil.format(new Date(), DatePattern.PURE_DATETIME_MS_PATTERN) + index;
        for (FmsErpTransactionEntity entity : eList) {
            entity.setBlendNo(blendNo);
            entity.setBlendType(FmsEnumType.BLEND_TYPE.BLEND_TYPE_2.getValue());
            entity.setLastReconciliationStatus(entity.getReconciliationStatus());
            entity.setReconciliationStatus(
                    FmsEnumType.RECONCILIATION_STATUS.RECONCILIATION_STATUS_3.getValue());
        }
        for (FmsTransactionDetailsHisDTO dto : bList) {
            dto.setBlendNo(blendNo);
        } 
    }
    
    if (!ruleFlag) {
        // 未匹配上数据 进入一下条
        continue;
    }
    index++;
}

# 第二步，银行唯一对账码、企业N条企业账和总金额相同自动对账规则; ruleConfig::oneToManyReconciliationCode == "Y"
for (Map.Entry<String, List<FmsTransactionDetailsHisDTO>> bankKey : bankMap.entrySet()) {
    String bankKeyStr = bankKey.getKey();
    # key为空直接进行下一次循环; key 不包含 ”,“ 直接进行下一次循环; 
    List<FmsTransactionDetailsHisDTO> detailsHisList = bankKey.getValue();
    # 通过key查询到的数据detailsHisList != 1 直接进行下一次循环; 
    #  detailsHisList.get(0).getBlendNo() 已生成勾兑套号, 直接进行下一次循环; 
    list = erpMap[bankKeyStr.split(",")]
    //唯一对账码和金额相等规则匹配表示 true-满足规则 false-不满足规则
    boolean ruleFlag = matchRule(ruleConfig, detailsHisList, list, index);
    {
        # matchRule 实现， 总金额、 交易类型 以及 交易时间 匹配
        String blendNo = DateUtil.format(new Date(), DatePattern.PURE_DATETIME_MS_PATTERN) + index;
        for (FmsErpTransactionEntity entity : eList) {
            entity.setBlendNo(blendNo);
            entity.setBlendType(FmsEnumType.BLEND_TYPE.BLEND_TYPE_2.getValue());
            entity.setLastReconciliationStatus(entity.getReconciliationStatus());
            entity.setReconciliationStatus(
                    FmsEnumType.RECONCILIATION_STATUS.RECONCILIATION_STATUS_3.getValue());
        }
        for (FmsTransactionDetailsHisDTO dto : bList) {
            dto.setBlendNo(blendNo);
        } 
    }
    if (!ruleFlag) {
        // 未匹配上数据 进入一下条
        continue;
    }
    index++;
    
}

# 第三步，银行流水有多个唯一对账码、银企账1个和总金额相同自动对账规则; ruleConfig::manyToOneReconciliationCode == "Y"
# busNo = [busMap.get(fmsBusinessNo)[busNo]]

erpMap  = erpList<reconciliationCode, []>
for key, erpTransactionEntities in erpMap:
