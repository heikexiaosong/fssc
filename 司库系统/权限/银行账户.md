# 银行账户
```sql

select t0.ID, t0.ORG_PERMISSION_TYPE, t0.SITES_TYPE, t0.MODE_ALL_FALG, t0.MENU_ALL_FALG, t0.LE_ALL_FALG, t0.ACCOUNT_ALL_FALG, t0.ORG_ALL_FALG, t0.BELONG_LE_ALL_FALG
  from fms_account_auth_config t0, fms_auth_config_emp t1, fms_auth_config_type t2
 where t0.id = t1.auth_config_id
   and t1.auth_config_id = t2.auth_config_id
   and t1.emp_id = '32094dc847076ccbeb863137322e26dd'
   and t2.auth_type = '1'
   and t0.enabled_flag = '0'
   and t0.validity_flag = '0'
group by t0.ID, t0.ORG_PERMISSION_TYPE, t0.SITES_TYPE, t0.MODE_ALL_FALG, t0.MENU_ALL_FALG, t0.LE_ALL_FALG, t0.ACCOUNT_ALL_FALG, t0.ORG_ALL_FALG, t0.BELONG_LE_ALL_FALG;


select t_ali.* 
  from ( select * 
           from fms_account fav
          where 1 = 1
            and validity_flag = '0'
            and enabled_flag = '0'
            and fav.group_id = 'decf31c11401b609562b3137322e000b'
            and bank_head_id = '102'
          limit 20
       ) t_ali
where 1 = 1
  and t_ali.bank_head_id = '102'
order by create_date desc, last_update_date, id desc
limit 20


select id, group_id, validity_flag, enabled_flag, bank_head_id   from fms_account where bank_head_id = '102'

```
