-- 放/还款数据关联查询
SELECT  r.id as '放款id', r.DISBURSEMENT_DETAIL_CODE as '放款编码', r.BOE_ID, r.BOE_NO as '放款单号', r.LOAN_ID as '内部贷款ID'
       , h.id as '还款申请单id',  h.BOE_NO as '还款申请单编码',  t.LINE_TYPE, t.id as '单据行id', t.PAY_AMOUNT as '申请金额'
       , r1.id as '还款id', r1.BOE_NO as '还款单号', r1.BOE_ID,  r1.LOAN_RECORD_ID as '放款登记ID', r1.AMOUNT
FROM zfs_fms.fms_inner_loan_ledger_record r
         left join zfs_foe.zfs_boe_lines t on t.PROVIDE_LOAN_ID = r.id
         left join zfs_foe.zfs_boe_header h on h.id = t.BOE_ID
         left join zfs_fms.fms_inner_loan_ledger_record r1 on r1.LOAN_RECORD_ID = r.id and r1.TYPE = 2    -- 放款登记ID = 放款id
WHERE r.LOAN_ID = 'ec93016d771d0f2bd7a63139322e22b4' and r.TYPE = 1
  and r1.AMOUNT = t.PAY_AMOUNT
  and r1.LOAN_RECORD_ID = '656e0d804b0cfb4b633a3139322e4426'
