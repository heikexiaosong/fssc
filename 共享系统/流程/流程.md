```json
{
  "boeNo": "",

```

#### 非结束单据流程查询
- https://127.0.0.1/fmc/flow/findWfHisTask
```json
{
  "boeNo": "GHG-AP2606150459",
  "groupId": "98384a38c6501ea777273132372e0008"
}
```

```sql
-- 1.  fmcWfBills
select *
  from FMC_WF_BILLS
 where BOE_CODE = 'GHG-AP2606150459'
   and GROUP_ID = '98384a38c6501ea777273132372e0008';


-- 2. 通过 流程模板id  fmcWfBills.getProcessId() 找到流程示例
select *
  from UFLO_PROCESS 
 where ID_ = :fmcWfBills.getProcessId();


-- 3. 查询历史任务记录
select *
  from FMC_WF_HIS_TASK_V 
 where BOE_CODE = 'GHG-AP2606150459'
  and GROUP_ID = '98384a38c6501ea777273132372e0008' and (assigneeName!='ACTION' or assigneeName is null;
 
```
