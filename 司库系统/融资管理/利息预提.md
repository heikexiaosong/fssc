```sql
select t_ali.*
from (
        select b.id, a.id as provide_header_id, a.PROVIDE_CODE, a.CONTRACT_NO, a.LE_ID
                , a.CURRENCY_CODE, a.PROVIDE_TYPE, a.PROVIDE_TERM, a.legal_person, a.legal_person_name
                , a.accounting_type
                , case
                        when c.FINANCIAL_INSTITUTION is not null then c.FINANCIAL_INSTITUTION
                        else a.BANK_ID
                end as BANK_ID
                , case
                        when c.CARDINALIY is not null then c.CARDINALIY
                        else a.CARDINALIY
                end as CARDINALIY, a.LOAN_CLIENT, a.LOAN_CLIENT_NAME, b.AMOUNT, b.ANNUAL_RATE
                , b.PROVISION_BEGIN_DATE, b.PROVISION_END_DATE, b.COUNT_DATE, b.PROVISION_INTEREST, b.last_update_by
                , b.last_update_date, b.CREATE_DATE, b.BOE_NO, b.IS_SYNC, b.EXTERNAL_SYSTEM_KEY
                , b.enabled_flag, b.validity_flag, b.provide_repay_id, b.auto_generated_flag, b.provide_loan_id
                , c.provide_loan_no as provide_loan_code
        from fms_provide_header a
                inner join fms_loan_provision_record b on a.id = b.provide_header_id -- 预提记录表
                left join fms_provide_loan c on c.id = b.provide_loan_id
        where b.enabled_flag = '0'
                and a.VALIDITY_FLAG = '0'
                and b.VALIDITY_FLAG = '0'
                and a.group_Id = 'ba2ad44f9aba2ad54fa43139322e0000'
) t_ali
where 1 = 1
        and t_ali.last_update_date >= timestamp '2026-03-06 00:00:00'
        and t_ali.last_update_date <= timestamp '2026-06-03 23:59:59.999'
        and (t_ali.provide_code like '%利随本清%')
order by CREATE_DATE desc
limit 20
```

```sql 
select t_ali.*
from (
        select distinct b.*
        from FMS_LOAN_RATE_COMPUTE a
                inner join FMS_PROVIDE_HEADER b on a.provide_Header_Id = b.id
                left  join FMS_PROVIDE_LOAN l on l.id = a.PROVIDE_LOAN_ID and l.ENABLED_FLAG = '0'
        where a.begin_date < TIMESTAMP '2026-06-04 00:00:00.000'
                and a.amount != 0 and a.amount is not null  -- 有效金额
                and a.annual_rate != 0 and a.annual_rate is not null -- 有效利率

                and b.enabled_Flag = '0'
                and b.validity_flag = '0'
                and b.group_Id = 'ba2ad44f9aba2ad54fa43139322e0000'
                and (   ( l.provision_date < a.end_date
                            and l.provision_date < TIMESTAMP '2026-06-04 00:00:00.000'
                            and a.provide_loan_id is not null)
                      or ( b.provision_date < a.end_date
                            and b.provision_date < TIMESTAMP '2026-06-04 00:00:00.000'
                            and a.provide_loan_id is null))
                and b.LE_ID in ('0b22741bcbaf4eb14b853139322e7d95')
) t_ali
where 1 = 1
        and t_ali.provide_code like '%利随本清%'
order by provide_code, Id asc
```
