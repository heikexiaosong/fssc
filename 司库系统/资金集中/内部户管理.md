        , t0.LE_ALL_FALG, t0.ACCOUNT_ALL_FALG, t0.ORG_ALL_FALG, t0.BELONG_LE_ALL_FALG

-- 余额查询
select a.*, ia.account_name, ia.bank_number, ia.le_id, ia.account_type
        , ia.overdraw_flag, ia.interest_flag, ia.account_state
from fms_inner_balance a
        inner join fms_inner_account ia on a.account_id = ia.id
where 1 = 1
        and exists (
                select fpr.le_id
                from fms_pool_org fpo
                        inner join fms_pool_reorg fpr on fpr.ORG_ID = fpo.id
                where fpo.ENABLED_FLAG = '0'
                        and fpo.VALIDITY_FLAG = '0'
                        and fpr.ENABLED_FLAG = '0'
                        and fpr.VALIDITY_FLAG = '0'
                        and fpr.le_id = ia.le_id
                        and (fpo.id in ('6f0f10ab765230d139d03137322e3206', 'ddf8cd7c07be1f03b0c83137322e1b37')
                                or fpo.FID like '/2cfff7a23ba225f47a773137322e1139/6f0f10ab765230d139d03137322e3206%')
        )
        and a.group_id = '98384a38c6501ea777273132372e0008'
        and a.enabled_flag = '0'
        and a.register_date >= TIMESTAMP '2026-01-01 00:00:00.000'
        and a.register_date < TIMESTAMP '2026-05-26 00:00:00.000'
        and ia.bank_number like '%0102201%'
        and (ia.cancel_date is null
                or a.register_date <= ia.cancel_date)
order by ia.bank_number, a.register_date desc
fetch first 20 rows only
