select *
from fms_provide_header fmsprovide0_
where fmsprovide0_.provide_code like '%2310202601100001187%'
        and fmsprovide0_.provide_status not in ('5')
        and fmsprovide0_.enabled_flag = '0'
        and fmsprovide0_.group_id = 'ba2ad44f9aba2ad54fa43139322e0000'
order by fmsprovide0_.provide_start_date desc, fmsprovide0_.id desc
limit 20
