```sql
select *
from fms_payment_instructions fmspayment0_
where (fmspayment0_.instruction_no like '%AP2605290005%' escape '\')
        and fmspayment0_.instruction_type = '1'
        and fmspayment0_.payment_status in ('12', '24', '13', '25', '40')
        and fmspayment0_.enabled_flag = '0'
        and fmspayment0_.group_id = '51561d61e6048d640e113139322e000d'
order by fmspayment0_.last_update_date desc, fmspayment0_.id desc
limit 20
```
