public Map<String, String> receiveOnePayResult(OpEfsPaymentDTO opEfsPaymentDTO, 
                                                   Map<String, OpEfsPaymentEntity> idEntityMap,
                                                   Map<String, OpEfsPaymentEntity> paymentNoEntityMap, 
                                                   Map<String, SysCompanyBankDTO> companyBankDTOMap,
                                                   Map<String, SimpleObjCacheDto> cacheDtoMap, 
                                                   List<OpEfsPaymentDTO> updatePaymentDTOs,
                                                   Map<String, String> remarkMap, 
                                                   Map<String, String> actionMap,
                                                   Map<String, String> boforeRequestParamMap, 
                                                   List<OpEfsPaymentDTO> opEfsPaymentlogs) {
    efsPaymentId = opEfsPaymentDTO.getId();
    paymentNo   = opEfsPaymentDTO.getPaymentNo();
    // one: 状态同步前 数据库指令信息
    one = idEntityMap[id=efsPaymentId] || paymentNoEntityMap[paymentNo=paymentNo]
    // 如果数据库指令状态=30, 不再执行更新逻辑， 退出
    if (one == null && one.getStatus() == 30) {
        return;
    }

    String boforeRequestParam = ZfsJsonUtil.toJsonStr(one);
    boforeRequestParamMap.put(id, boforeRequestParam);
    // TODO one 设置字段值 ...
    OpEfsPaymentDTO operEfsPayDto = ConvertUtils.sourceToTarget(one, OpEfsPaymentDTO.class);
    updatePaymentDTOs.add(operEfsPayDto);
    
    opEfsPaymentlogs.add(operEfsPayDto);

    // 银行流水号
    remarkMap.put(id, opEfsPaymentDTO.getBankFlowNumber());

    // 保存操作记录
    String operAction = opEfsPaymentDTO.getOperAction();
    if(CharSequenceUtil.isBlank(operAction)){
        operAction = EnumType.PaymentAction.ReceiveStatus.name();
    }
    actionMap.put(id, operAction);
    
                                                       
}

public void updateDetail(List<OpEfsPaymentDTO> opEfsPaymentDTOS) {
    logger.info("保存支付数据opEfsPaymentDTOS入参:{}",opEfsPaymentDTOS.size());

    idList             = [opEfsPaymentDTOS::id]
    opEfsPaymentDtosDb = OP_EFS_PAYMENT[id in :idList]
    opEfsPaymentMapDb  = opEfsPaymentDtosDb.groupBy { it.id }

    // 拿重生成支付数据之前的付款id
    preEfsPaymentIdSet = [opEfsPaymentDTOS::preEfsPaymentId]
    opEfsPaymentHisEntityMap = OP_EFS_PAYMENT_HIS[id in :preEfsPaymentIdSet].groupBy { it.id }

    for (OpEfsPaymentDTO opEfsPaymentDTO : opEfsPaymentDTOS) {
        String efsId = opEfsPaymentDTO.getId();
        // 当前数据库中指令信息
        OpEfsPaymentEntity storeEntity  = opEfsPaymentMapDb.get(efsId);
        
        // 最新指令信息 复制到 entity中    
        EntityUtils.copyPageToDb(opEfsPaymentDTO, storeEntity);
