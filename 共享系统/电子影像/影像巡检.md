SELECT EI.id, EI.id AS image_id, EI.image_name, EI.image_path, EI.thumbnail_path, EI.is_show_image AS img_sec, EI.order_num AS img_order, EI.classify_count AS seg_counts, EI.image_info_id,
        EII.image_number
  FROM EID_IMAGE EI
        JOIN EID_IMAGE_INFOS EII ON EII.ID = EI.IMAGE_INFO_ID
        JOIN EVS_OPERATION_TYPES EOT ON EOT.ID = EII.OPERATION_TYPE_ID AND EOT.GROUP_ID = EII.group_id
 WHERE EI.OCR_FLAG IN ('N', 'F')
        AND EI.IMAGE_SOURCE_SYSTEM != 'FSSC'
        AND EI.IMAGE_SOURCE_SYSTEM is not null
        AND EII.IMAGE_PROCESS IN ('01', '02')

            AND EII.UPDATED_DT <![CDATA[   >=  ]]> #{updateDate}
        </if>

-- ids: [ocrImages::imageInfoId]
SELECT EII.ID, EII.ID AS IMAGE_INFO_ID, EII.IMAGE_PROCESS, EII.GROUP_ID, EII.IMAGE_NUMBER, EII.IMAGE_COUNT, EII.IMAGE_STATUS, EII.UPLOAD_DATE, EII.UPLOAD_EMPLOYEE_ID,
        (select look.meaning
           from evs_lookup_values look
          where look.group_id = EII.group_id
            and look.lookup_code = EII.image_status
            and look.lookup_type = 'IMG_STATUS_TYPE' and look.enabled = 'Y') as STATUS,
        (select o.operation_type_code 
           from evs_operation_types o

  FROM EID_IMAGE_INFOS EII

        AND EII.IMAGE_STATUS IN ('COMPLEMENT','HEAVY_AND','UPLOADING','AUDITED')

        AND EII.SOURCE_CODE != 'EAS' AND EII.SOURCE_CODE is not null
