{

```sql
-- 相关SQL
-- 1. 获取还息计划信息
select *
from FMS_PROVIDE_REPAY_PLAN
where PROVIDE_ID = '7138bbd96d42449e023d3139322e5f34'
      and REPAY_TYPE = '2'
      and enabled_flag = '0'
```
