## 回单匹配流水

### 电子回单匹配流水 
- cn.ztessc.service.task.JobCenter#synEleMatchDetails
```sql
-- 每次匹配 100 条
select * 
  from FMS_ELECTRONIC_RECEIPT 
 where ENABLED_FLAG = '0' 
   and (BATCH_NUMBER_SYS is null or BATCH_NUMBER_SYS = '')
   and (DETAIL_MATCH_TIME <= 50 or DETAIL_MATCH_TIME is null ) 
order by DETAIL_MATCH_TIME asc,CREATE_DATE desc
```

  from FMS_TRANSACTION_DETAILS_HIS 
 where TRANSACTION_DATE >= :beginTransactionDate
   and TRANSACTION_DATE <= :endTransactionDate
   and BILL_RECEIPT_NO is null 
   and ENABLED_FLAG = '0' 
order by create_date asc, id desc

-- batchNumberSysList: [FmsTransactionDetailsHisEntity::id] 流水id

-- 1. 查询银行下的可同步回单的账号数量
select count(1)
from fms_account fav
where 1 = 1
  and fav.E_BANK_MONOBLOCK = '0'
  and fav.SALES_MARK <> '1'
  and fav.enabled_flag = '0'
  and fav.validity_flag = '0'
  and fav.BANK_HEAD_ID in ('203')

-- 2. 查询账号对应的回单记录
select *
  from fms_electronic_receipt
 where (payer_bank_account_num in (:bankAccountNumList) or gather_bank_account_num in (:bankAccountNumList))
   and enabled_flag = '0'
   and (fmselectro0_.sync_file_url is null or fmselectro0_.sync_file_url= '') 
   and transaction_date>= '2026-05-18 00:00:00' 
   and transaction_date<= '2026-05-20 00:00:00'
