#### 资金归集明细数据查询接口 (FSSC-API-000175)
- /outapi/queryFinanceCollectDetails

```sql
-- 仅查询支付状态为付款成功的数据 paymentStatus = '30'
-- collectInstructionsDtos
SELECT * 
  FROM FMS_COLLECT_INSTRUCTIONS  -- 归集指令表

SELECT * 
  FROM FMS_PAYMENT_APPLY_HEADER  -- 付款申请单头表
 where id in (collectInstructionsDtos::BOE_ID)


 -- 根据外部账号ID查询对应的内部账号ID
 -- outAccountIds = [collectInstructionsDtos::payerBankAccountId, collectInstructionsDtos::gatherBankAccountId]
 SELECT t1.* 
   FROM FMS_ACCOUNT_MAPPING t1 
  WHERE WHERE t1.OUT_ACCOUNT_ID in :outAccountIds

-- relationMap: FMS_ACCOUNT_MAPPING<OUT_ACCOUNT_ID, INNER_ACCOUNT_ID>

-- 内部户信息
-- innerAccountIds: [FMS_ACCOUNT_MAPPING::INNER_ACCOUNT_ID]
SELECT t1.* 
  FROM FMS_INNER_ACCOUNT t1
 WHERE t1.ID IN (:innerAccountIds)

 -- innerAccountDtosMap： FMS_INNER_ACCOUNT<ID, FMS_INNER_ACCOUNT>

 

```

select b.ID, a.BANK_NUMBER, a.ACCOUNT_NAME, a.LE_ID, b.COLLECT_TYPE
        , b.inner_account_id, b.CURRENCY_CODE, b.OPPOSITE_BANK_NUMBER, b.OPPOSITE_BANK_NAME, b.INTERIOR_DETAIL_STATUS
        , b.GENERATE_NODE, b.BUSINESS_DATE, b.REMARK, b.DEBIT_AMOUNT, b.CREDIT_AMOUNT
        , b.BUSINESS_BOE_NO, b.GROUP_ID, b.LAST_UPDATE_DATE, b.IS_SYNC, b.TRANSACTION_ID
        , b.DEBIT_FLAG, b.TRANS_BALANCE, b.TRANSACTION_PURPOSE, b.SOURCE, b.CREATE_BY
        , b.CREATE_DATE, b.BOE_NO, b.EXTERNAL_SYSTEM_KEY, a.CONTACT_USER_ID, a.ORG_ID
        , b.SOURCE_SYSTEM, b.BOE_TYPE_CODE, b.TASK_ID, b.BUSINESS_TYPE, b.FORM_TYPE_ID
        , b.MAPPED_TYPE, b.NEW_BILL_STATUS, b.BILL_TYPE_CODE, b.BUS_ID, b.INTEREST_TAX_RATE
        , b.INTEREST_TAX_AMOUNT, b.INTEREST_EXCLUSIVE_AMOUNT, a.BANK_NUMBER as INNER_BANK_NUMBER, a.ACCOUNT_NAME as INNER_ACCOUNT_NAME, b.UNIQUE_RECONCILIATION_CODE
        , case
                when b.MATCH_STATUS is not null then b.MATCH_STATUS
                else '0'
        end as MATCH_STATUS, b.UNIQUE_NO
from FMS_INNER_ACCOUNT a
        inner join FMS_INTERIOR_DETAIL b on a.ID = b.INNER_ACCOUNT_ID
where 1 = 1
        and b.business_date >= TIMESTAMP '2026-05-01 00:00:00.000'
        and b.business_date <= TIMESTAMP '2026-05-22 23:59:59.999'
        and b.business_boe_no like '%212605221292%'
        and b.ENABLED_FLAG = '0'
        and b.GROUP_ID = 'ba2ad44f9aba2ad54fa43139322e0000'
order by BANK_NUMBER asc, business_date asc, last_update_date asc, id asc
limit 20
