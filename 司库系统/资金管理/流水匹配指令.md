### 交易明细匹配指令
cn.ztessc.service.task.JobCenter#synDetailMatchInstruction
```sql
-- 
select * from fms_account fav where 1=1

-- 当日待匹配数据
select * 
from FMS_TRANSACTION_DETAILS_DAY 
where ENABLED_FLAG = '0' 
  and ACCOUNT_ID = :accountId 
  and INSTRUCTION_NO is null 
  and TRANSACTION_TYPE = '0' 
  and INSTRUCTION_SOURCE is null 
  and BANK_NUMBER is not null
  and (INSTRUCTION_MATCH_TIME <= 50 or INSTRUCTION_MATCH_TIME is null ) 
order by INSTRUCTION_MATCH_TIME asc,CREATE_DATE desc

-- key: bankNumber^currencyCode^transactionAmount
-- batchConditionMap： <key, [<"conditionBeginDate", TransactionDate-DiffTIme>, <"conditionEndDate", TransactionDate+DiffTIme>]>


-- 
select i.* 
  from fms_payment_instructions i 
       left join FMS_BATCH_QUERY_CONDITION c on c.PAYER_BANK_ACCOUNT_NUM=i.PAYER_BANK_ACCOUNT_NUM and c.CURRENCY_CODE=i.PAYMENT_CURRENCY and c.CONDITION_AMOUNT=i.APPLY_AMOUNT
 where i.ENABLED_FLAG = '0' 
  and i.PAYMENT_STATUS = '30' 
  and i.INSTRUCTION_TYPE in ('1', '3')
  and c.CONDITION_TYPE = '002' -- 交易明细匹配指令 cn.ztessc.utils.FmsEnumType.conditionType#CONDITION_TYPE_002 
  and c.CONDITION_BATCH_NO = :conditionBatchNo
  and i.PAYMENT_DATE >= c.CONDITION_BEGIN_DATE 
  and i.PAYMENT_DATE <= c.CONDITION_END_DATE 

-- 更新流水数据
update fms_transaction_details_day 
set INSTRUCTION_NO = ? , INSTRUCTION_MATCH_TIME = ? , INSTRUCTION_SOURCE = ? , PAYMENT_FLOW_FLAG = ?
where id = ? and INSTRUCTION_NO is null

```
