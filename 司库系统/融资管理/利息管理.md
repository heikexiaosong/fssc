```sql
-- 相关SQL
-- 1. 获取放款信息
SELECT b.id, b.provide_code AS provide_loan_code, b.loan_amount, b.loan_date, b.financial_institution, b.cardinaliy, b.annual_rate, b.provide_header_id, a.provide_code, a.contract_no, a.le_id, a.legal_person, a.legal_person_name, a.bank_id, a.currency_code, a.provide_type, a.accounting_type, a.provide_amount, a.provide_start_date, a.provide_end_date, a.provide_term, a.provide_status
from FMS_PROVIDE_HEADER a
 inner join FMS_PROVIDE_LOAN b on b.provide_header_id = a.id
where a.enabled_flag = '0'
  and a.validity_flag = '0'
  and b.enabled_flag = '0'
  and b.validity_flag = '0'
  and b.PROVIDE_LOAN_NO like '%2023年工银金租境内综合一字第004号-ZLFB%';


-- 2. 获取放款信息的利率调整记录
select *
from FMS_LOAN_INTEREST_RECORD
where PROVIDE_LOAN_ID = '332fa8f203147cd204ae3139322e5863'
and enabled_flag = '0'

-- 3. 获取放款信息的LPR调整日期列表（LPR_ADJUST_DATE）
select *
from FMS_PROVIDE_LPR_DATE sd
where sd.PROVIDE_ID = '332fa8f203147cd204ae3139322e5863'
  and sd.enabled_flag = '0'
```
