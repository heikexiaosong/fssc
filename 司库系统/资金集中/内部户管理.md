```sql
-- 权限查询
select t0.ID, t0.ORG_PERMISSION_TYPE, t0.SITES_TYPE, t0.MODE_ALL_FALG, t0.MENU_ALL_FALG
        , t0.LE_ALL_FALG, t0.ACCOUNT_ALL_FALG, t0.ORG_ALL_FALG, t0.BELONG_LE_ALL_FALG
from fms_account_auth_config t0, fms_auth_config_emp t1, fms_auth_config_type t2
where t0.id = t1.auth_config_id
        and t1.auth_config_id = t2.auth_config_id
        and t1.emp_id = '36881a8cf915dd72f5b93137322e0da4'
        and t2.auth_type = '13'
        and t0.enabled_flag = '0'
        and t0.validity_flag = '0'
group by t0.ID, t0.ORG_PERMISSION_TYPE, t0.SITES_TYPE, t0.MODE_ALL_FALG, t0.MENU_ALL_FALG, t0.LE_ALL_FALG, t0.ACCOUNT_ALL_FALG, t0.ORG_ALL_FALG, t0.BELONG_LE_ALL_FALG

-- 余额查询
select a.*, ia.account_name, ia.bank_number, ia.le_id, ia.account_type
        , ia.overdraw_flag, ia.interest_flag, ia.account_state
from fms_inner_balance a
        inner join fms_inner_account ia on a.account_id = ia.id
where 1 = 1
        and exists (
                select fpr.le_id
                from fms_pool_org fpo
                        inner join fms_pool_reorg fpr on fpr.ORG_ID = fpo.id
                where fpo.ENABLED_FLAG = '0'
                        and fpo.VALIDITY_FLAG = '0'
                        and fpr.ENABLED_FLAG = '0'
                        and fpr.VALIDITY_FLAG = '0'
                        and fpr.le_id = ia.le_id
                        and (fpo.id in ('6f0f10ab765230d139d03137322e3206', 'ddf8cd7c07be1f03b0c83137322e1b37')
                                or fpo.FID like '/2cfff7a23ba225f47a773137322e1139/6f0f10ab765230d139d03137322e3206%')
        )
        and a.group_id = '98384a38c6501ea777273132372e0008'
        and a.enabled_flag = '0'
        and a.register_date >= TIMESTAMP '2026-01-01 00:00:00.000'
        and a.register_date < TIMESTAMP '2026-05-26 00:00:00.000'
        and ia.bank_number like '%0102201%'
        and (ia.cancel_date is null
                or a.register_date <= ia.cancel_date)
order by ia.bank_number, a.register_date desc
fetch first 20 rows only
```

```sql
-- 获取内部户信息 accountEntity
select *
from FMS_INNER_ACCOUNT
where id = '4ef6263bec949039564d3137322e0b4c'

-- throw Exception(date < accountEntity:lastInterestDate>)

-- 获取日期内内部户余额积数最大日期记录 initAmount
select t.* 
  from fms_inner_balance  t 
    right join (
      select ACCOUNT_ID,max(REGISTER_DATE) as REGISTER_DATE from fms_inner_balance fib where 1=1
      and fib.ACCOUNT_ID = :accountId
      and fib.REGISTER_DATE < :startDate
      and fib.enabled_flag = '0'
      group by fib.ACCOUNT_ID ) tmp on t.ACCOUNT_ID=tmp.ACCOUNT_ID and t.REGISTER_DATE= tmp.REGISTER_DATE
  where t.ACCOUNT_ID = :accountId
    and t.enabled_flag = '0'

-- 判断开始日期是否是余额日期的下一天，不是下一天要累加进来
-- accountCurrencyListByAccountId
select *
  from FMS_INNER_CURRENCY
 where ACCOUNT_ID = :accountId
   and enabled_flag = '0'


-- 查询已存在的余额积数
-- afterList   ->  map: key=registerDateNum, value=balance
select *
  from FMS_INNER_BALANCE
 where ACCOUNT_ID = :accountId
   and REGISTER_DATE >= :startDate
   and REGISTER_DATE < :endDate
   and enabled_flag = '0'
```
