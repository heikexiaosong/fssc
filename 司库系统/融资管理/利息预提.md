-- 查询 贷款信息
select t_ali.*
from (
        select distinct b.*
        from FMS_LOAN_RATE_COMPUTE a
                inner join FMS_PROVIDE_HEADER b on a.provide_Header_Id = b.id
                left join fms_provide_loan l on l.id = a.PROVIDE_LOAN_ID and l.enabled_Flag = '0'
        where a.begin_date < TIMESTAMP '2026-05-19 00:00:00.000'
                and a.amount != 0
                and a.amount is not null
                and a.annual_rate != 0
                and a.annual_rate is not null
                and b.provide_status in ('2', '3', '7')
                and b.enabled_Flag = '0'
                and b.validity_flag = '0'
                and b.group_Id = 'ba2ad44f9aba2ad54fa43139322e0000'
                and ((a.end_date > l.provision_date
                                and l.provision_date < TIMESTAMP '2026-05-19 00:00:00.000'
                                and a.provide_loan_id is not null)
                        or (a.end_date > b.provision_date
                                and b.provision_date < TIMESTAMP '2026-05-19 00:00:00.000'
                                and a.provide_loan_id is null))
) t_ali
where 1 = 1
        and t_ali.provide_code like '%0502测试%'
order by provide_code, Id asc
limit 20
