-- 查询流水状态
select *
from claim_flow_summary claimflows0_
where claimflows0_.last_update_date >= TIMESTAMP '2026-03-31 00:00:00.000'
        and claimflows0_.last_update_date < TIMESTAMP '2026-03-31 14:00:00.000'
        and claimflows0_.claim_status in ('60', '50')
        and claimflows0_.enabled_flag = '0'
        and claimflows0_.group_id = '43126ac934a1b0007955b1b9b2ae0005'
order by claimflows0_.act_date desc, claimflows0_.create_date desc

from claim_boe_record claimboere0_
where claimboere0_.claim_id in ('c07902d59c3684a329c431302e320357', 'a28ab97476929d9d79bc31302e3249bc', 'c07902d59c3684a5cfff31302e320358', 'a28ab9747692a6c6aa3931302e324c1f', 'a28ab974a13fff72109b31302e3256e6', 'c07902d5a2f9a91ee81f31302e327e33', 'a28ab9747692a99d39bd31302e324d0b', 'a28ab9747692b1ebe14431302e324fe8', 'a28ab974a14014163dd831302e325cd0', 'a28ab97476929dade58731302e3249c1')
        and claimboere0_.enabled_flag = '0'
        and claimboere0_.validity_flag = '0'


-- 排查问题数据 SQL: 应该关联未关联数据    
select s.id, s.HOSTID, s.CLAIM_STATUS , s.LAST_UPDATE_DATE, r.BOE_NO, r.BOE_STATUS , h.ID, h.FMS_BUSINESS_NO, b.BUS_ID , b.BUS_NO 
from zfs_claim.claim_boe_record r
     left join zfs_claim.claim_flow_summary s on s.ID  = r.CLAIM_ID 
     left join zfs_fms.fms_transaction_details_his h on h.TRANSACTION_NO  = s.HOSTID 
     left join zfs_fms.fms_transaction_details_bus b on b.FMS_BUSINESS_NO = h.FMS_BUSINESS_NO 
where 1=1
  and b.BUS_ID is null 
order by s.LAST_UPDATE_DATE desc
